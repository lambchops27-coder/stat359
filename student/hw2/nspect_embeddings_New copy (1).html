# Creating plot format 
def tsne_plot(name, get_vec_fn):
    X, labels, words = collect_vectors(get_vec_fn)

    tsne = TSNE(n_components=2, random_state = 0, perplexity =10)
    Y = tsne.fit_transform(X)

    plt.figure(figsize=(7, 7))
    for lbl in ['animal', 'food']:
        idx = [i for i, l in enumerate(labels) if l == lbl]
        plt.scatter(Y[idx,0], Y[idx,1], label=lbl)

    for i, w in enumerate(words):
        plt.text(Y[i,0], Y[i,1], w, fontsize = 9)

    plt.title(f"t-SNE: {name}")
    plt.legend()
    plt.show()

def collect_vectors(get_vec_fn):
    words = animals + foods
    X = []
    labels = []
    kept_words = []

    for w in words:
        v = get_vec_fn(w)
        if v is not None:
            X.append(v)
            kept_words.append(w)
            labels.append("animal" if w in animals else "food")

    return np.array(X), labels, kept_words

# Create umap object
def umap_plot(title, get_vec_fn):
    words = foods + animals
    kept_words = []
    vectors = []
    colors = []
    for w in words:
        v = get_vec_fn(w)
        if v is not None:
            kept_words.append(w)
            vectors.append(v)
            if w in foods:
                colors.append('orange')
            else:
                colors.append('blue')
    vectors = np.array(vectors, dtype = np.float32)
    reducer = umap.UMAP(n_neighbors = 10,
    min_dist = 0.1,
    n_components = 2,
    random_state = 42)
    reduced = reducer.fit_transform(vectors)
    plt.figure(figsize = (8, 6))
    for i, w in enumerate(kept_words):
        plt.scatter(reduced[i, 0], reduced[i, 1], c=colors[i])
        plt.annotate(w, (reduced[i, 0], reduced[i, 1]))

    plt.title(title)
    plt.xlabel('UMAP 1')
    plt.ylabel('UMAP 2')
    plt.show()

Short answer questions

Which embeddings performed best at capturing sentence meaning?

Pretrained Fasttext has the best performance capturing sentence meaning in both TSNE and UMAP. This is because it has a relatively accurate distinction between animals and animal meat products. Furthermore, when grouping animals in their relative position, similar animals are groups relatively proximate to one another, e.g. in TSNE and UMAP - whale, dolphin and sharks, all large marine animals or banana, orange and apple, all fruits are grouped together.

Were the pretrained embeddings significantly better than your own? Why or why not?

The pretrained embeddings are indeed significantly better than Pytorch trained embeddings. Evident in the plots, Pytorch model shows much weaker clustering between words similar to each other and a weaker semantic structure, where animals and fruits were clustered together. This is rather expected, given that pretrained models are trained on corpa of texts much larger and diverse in source, using Wikipedia and Google news articles, alongside a longer training period with richer contexts and relatively stable optimization.

What factors might explain the differences in performance?

A key factor leading to this difference is the corpus size each embedding is trained. Pretrained models were trained on corpa that were large, with billions of tokens, alongside a diverse context and source. The Pytorch model contrarily was only trained on Text 8. Another factor is Fasttext's incorporation of subword information enabling its effective representation of rare and morphologically-close/related words. Furthermore, pretrained models have extensive parameter tuning and longer training schedules.
Lastly, the weaknesses of underlying word vectors are amplified when averaging word embeddings for sentence representations, and this disproportionately impacted smaller or noisier models.
